458. Last Position of Target
public class Solution{
    public int lastPosition(int[] nums, int target) {
        int start = 0, end = nums.length - 1, index = -1;
        while (start <= end) {
            int mid = start + (end - start)/2;
            if (nums[mid] == target) {
                index = mid;
            }
            if (target < nums[mid]) {
                end = mid - 1;
            } else {
                start = mid + 1;
            }
        }
        return index;
    }
}
